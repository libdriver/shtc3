48uint8_t shtc3_basic_init(void)
49{
50 uint8_t res;
51
52 /* link functions */
53 DRIVER_SHTC3_LINK_INIT(&gs_handle, shtc3_handle_t);
54 DRIVER_SHTC3_LINK_IIC_INIT(&gs_handle, shtc3_interface_iic_init);
55 DRIVER_SHTC3_LINK_IIC_DEINIT(&gs_handle, shtc3_interface_iic_deinit);
56#if (SHTC3_BASIC_DEFAULT_CLOCK_STRETCHING != 1)
57 DRIVER_SHTC3_LINK_IIC_READ_ADDRESS16(&gs_handle, shtc3_interface_iic_read_address16);
58#else
59 DRIVER_SHTC3_LINK_IIC_READ_ADDRESS16(&gs_handle, shtc3_interface_iic_scl_read_address16);
60#endif
61 DRIVER_SHTC3_LINK_IIC_WRITE_ADDRESS16(&gs_handle, shtc3_interface_iic_write_address16);
62 DRIVER_SHTC3_LINK_DELAY_MS(&gs_handle, shtc3_interface_delay_ms);
63 DRIVER_SHTC3_LINK_DEBUG_PRINT(&gs_handle, shtc3_interface_debug_print);
64
65 /* shtc3 init */
66 res = shtc3_init(&gs_handle);
67 if (res != 0)
68 {
69 shtc3_interface_debug_print("shtc3: init failed.\n");
70
71 return 1;
72 }
73
74 return 0;
75}
